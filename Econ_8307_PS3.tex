	\begin{euqation}
	\begin{split}
	M(1-\delta)+\epsilon&=M\\

	\end{equation}

	Denote the number of firm of type $z_i$ at time $t$ as $\mu_t(z_i)$
	\begin{equation}
	\mu_{t+1}(z^')=(1-\delta)\sum_{z}[f(z^'|z)\mu_t(z)]+\epsilon \phi(z^')
	\end{equation}
	\item
	\begin{equation}
	\begin{split}
	(1-\delta)T^'\mu&=-\epsilon \mu\\
	\mu&=-\epsilon((1-\delta)T^'-I)^{-1}\phi
	\end{split}
	\end{equation}
	\item
	\begin{lstlisting}
N=10;
epsilon=1;
delta=.1;
beta=1;
gamma=.1;
z=1:N;
transitMatrix=eye(N);
alpha=zeros(N,1);
phi=ones(N,1)/N;
mu=zeros(N,1);

for i=1:N
	for j=1:N
		alpha(i)=alpha(i)+max(0,beta-gamma*(z(j)-z(i))^2);
	end
end
for i=1:N
	for j=1:N
		transitMatrix(i,j)=max(0,(beta-gamma*(z(j)-z(i))^2)/alpha(i));
	end
end

mu=((1-delta)*transitMatrix.'-eye(N))\(-epsilon*phi)
	\end{lstlisting}
